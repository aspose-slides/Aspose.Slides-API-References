>  try {
>      ISlide slide = presentation.getSlides().get_Item(0);
> 
>      // Holt die Effektsequenz für die Folie
>      ISequence effectsSequence = slide.getTimeline().getMainSequence();
> 
>      for (IEffect effect : effectsSequence)
>      {
>          if (effect.getSound() == null)
>              continue;
>
