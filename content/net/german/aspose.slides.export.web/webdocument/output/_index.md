### Beispiele

```csharp
[C#]
using (Presentation pres = new Presentation("pres.pptx"))
{
    var options = new WebDocumentOptions
    {
        TemplateEngine = new RazorTemplateEngine(),
        OutputSaver = new FileOutputSaver(),
        EmbedImages = false
    };
    
    WebDocument document = new WebDocument(options);

    // "index.html" zu den Ausgabedateien hinzufügen, unter Verwendung der "index"-Vorlage zur Generierung und der pres-Variable als Modell
    document.Output.Add("index.html", "index", pres);

    // ... weitere Optionen des Dokuments einrichten und dann das Dokument speichern
    document.Save();
}
```
