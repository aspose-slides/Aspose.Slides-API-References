[C#]
using (Presentation presentation = new Presentation("demo.pptx"))
{
    ISlide slide = presentation.Slides[0];
    
    // Obtém a sequência de efeitos para o slide
    ISequence effectsSequence = slide.Timeline.MainSequence;
