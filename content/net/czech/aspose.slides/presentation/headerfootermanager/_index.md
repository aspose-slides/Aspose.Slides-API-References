### Příklady

Následující příklad ukazuje, jak nastavit viditelnost zápatí uvnitř Slide PowerPoint Presentation.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IBaseSlideHeaderFooterManager headerFooterManager = presentation.Slides[0].HeaderFooterManager;
	// Vlastnost IsFooterVisible se používá k indikaci, že zástupný prvek zápatí snímku není přítomen.
    if (!headerFooterManager.IsFooterVisible)
    {
		// Metoda SetFooterVisibility se používá k tomu, aby byl zástupný prvek zápatí snímku viditelný.
        headerFooterManager.SetFooterVisibility(true);
    }
	// Vlastnost IsSlideNumberVisible se používá k indikaci, že zástupný prvek čísla stránky snímku není přítomen.
    if (!headerFooterManager.IsSlideNumberVisible)
    {
		// Metoda SetSlideNumberVisibility se používá k tomu, aby byl zástupný prvek čísla stránky snímku viditelný.
        headerFooterManager.SetSlideNumberVisibility(true);
    }
	// Vlastnost IsDateTimeVisible se používá k indikaci, že zástupný prvek data a času snímku není přítomen.
    if (!headerFooterManager.IsDateTimeVisible)
    {
		// Metoda SetFooterVisibility se používá k tomu, aby byl zástupný prvek data a času snímku viditelný.
        headerFooterManager.SetDateTimeVisibility(true);
    }
	// Metoda SetFooterText se používá k nastavení textu do zástupného prvku zápatí snímku.
    headerFooterManager.SetFooterText("Footer text");
	// Metoda SetDateTimeText se používá k nastavení textu do zástupného prvku data a času snímku.
    headerFooterManager.SetDateTimeText("Date and time text");
	presentation.Save("Presentation.ppt",SaveFormat.ppt);
}
```
