### الأمثلة

يوضح المثال التالي كيفية تعيين رؤية التذييل داخل Slide من PowerPoint Presentation.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IBaseSlideHeaderFooterManager headerFooterManager = presentation.Slides[0].HeaderFooterManager;
	// الخاصية IsFooterVisible تُستَخدم للإشارة إلى أن عنصر نائب تذييل الشريحة غير موجود.
    if (!headerFooterManager.IsFooterVisible)
    {
		// الطريقة SetFooterVisibility تُستَخدم لجعل عنصر نائب تذييل الشريحة مرئياً.
        headerFooterManager.SetFooterVisibility(true);
    }
	// الخاصية IsSlideNumberVisible تُستَخدم للإشارة إلى أن عنصر نائب رقم صفحة الشريحة غير موجود.
    if (!headerFooterManager.IsSlideNumberVisible)
    {
		// الطريقة SetSlideNumberVisibility تُستَخدم لجعل عنصر نائب رقم صفحة الشريحة مرئياً.
        headerFooterManager.SetSlideNumberVisibility(true);
    }
	// الخاصية IsDateTimeVisible تُستَخدم للإشارة إلى أن عنصر نائب تاريخ-وقت الشريحة غير موجود.
    if (!headerFooterManager.IsDateTimeVisible)
    {
		// الطريقة SetFooterVisibility تُستَخدم لجعل عنصر نائب تاريخ-وقت الشريحة مرئياً.
        headerFooterManager.SetDateTimeVisibility(true);
    }
	// الطريقة SetFooterText تُستَخدم لتعيين النص إلى عنصر نائب تذييل الشريحة.
    headerFooterManager.SetFooterText("Footer text");
	// الطريقة SetDateTimeText تُستَخدم لتعيين النص إلى عنصر نائب تاريخ-وقت الشريحة.
    headerFooterManager.SetDateTimeText("Date and time text");
	presentation.Save("Presentation.ppt",SaveFormat.ppt);
}
```
