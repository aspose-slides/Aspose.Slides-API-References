### Exempel

Följande exempel visar hur man ställer in sidfotens synlighet i en Slide i en PowerPoint-presentation.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IBaseSlideHeaderFooterManager headerFooterManager = presentation.Slides[0].HeaderFooterManager;
	// Egenskapen IsFooterVisible används för att indikera att en sidfotplatshållare saknas.
    if (!headerFooterManager.IsFooterVisible)
    {
		// Metoden SetFooterVisibility används för att göra en sidfotplatshållare synlig.
        headerFooterManager.SetFooterVisibility(true);
    }
	// Egenskapen IsSlideNumberVisible används för att indikera att en sidnummerplatshållare saknas.
    if (!headerFooterManager.IsSlideNumberVisible)
    {
		// Metoden SetSlideNumberVisibility används för att göra en sidnummerplatshållare synlig.
        headerFooterManager.SetSlideNumberVisibility(true);
    }
	// Egenskapen IsDateTimeVisible används för att indikera att en datum-tid-platshållare saknas.
    if (!headerFooterManager.IsDateTimeVisible)
    {
		// Metoden SetFooterVisibility används för att göra en datum-tid-platshållare synlig.
        headerFooterManager.SetDateTimeVisibility(true);
    }
	// Metoden SetFooterText används för att sätta text i en sidfotplatshållare.
    headerFooterManager.SetFooterText("Footer text");
	// Metoden SetDateTimeText används för att sätta text i en datum-tid-platshållare.
    headerFooterManager.SetDateTimeText("Date and time text");
	presentation.Save("Presentation.ppt",SaveFormat.ppt);
}
```
