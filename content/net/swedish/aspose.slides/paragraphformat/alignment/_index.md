### Exempel

Följande exempelprogram visar hur man justerar textstycken i en PowerPoint-presentation.

```csharp
[C#]
// Skapa ett Presentation-objekt som representerar en PPTX-fil
using (Presentation pres = new Presentation("ParagraphsAlignment.pptx"))
{
    // Hämtar den första bilden
    ISlide slide = pres.Slides[0];
    // Hämtar den första och andra platshållaren i bilden och kastar den till AutoShape
    ITextFrame tf1 = ((IAutoShape)slide.Shapes[0]).TextFrame;
    ITextFrame tf2 = ((IAutoShape)slide.Shapes[1]).TextFrame;
    // Ändra texten i båda platshållarna
    tf1.Text = "Center Align by Aspose";
    tf2.Text = "Center Align by Aspose";
    // Hämtar det första stycket i platshållarna
    IParagraph para1 = tf1.Paragraphs[0];
    IParagraph para2 = tf2.Paragraphs[0];
    // Justera textstycket till mitten
    para1.ParagraphFormat.Alignment = TextAlignment.Center;
    para2.ParagraphFormat.Alignment = TextAlignment.Center;
    //Skriver presentationen som en PPTX-fil
    pres.Save("Centeralign_out.pptx", SaveFormat.Pptx);
}
```
