### Voorbeelden

Het volgende voorbeeld laat zien hoe de voettekst-zichtbaarheid in een Slide van een PowerPoint-presentatie kan worden ingesteld.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IBaseSlideHeaderFooterManager headerFooterManager = presentation.Slides[0].HeaderFooterManager;
	// Eigenschap IsFooterVisible wordt gebruikt om aan te geven dat een slide-voettekst-placeholder niet aanwezig is.
    if (!headerFooterManager.IsFooterVisible)
    {
		// Methode SetFooterVisibility wordt gebruikt om een slide-voettekst-placeholder zichtbaar te maken.
        headerFooterManager.SetFooterVisibility(true);
    }
	// Eigenschap IsSlideNumberVisible wordt gebruikt om aan te geven dat een slide-paginanummer-placeholder niet aanwezig is.
    if (!headerFooterManager.IsSlideNumberVisible)
    {
		// Methode SetSlideNumberVisibility wordt gebruikt om een slide-paginanummer-placeholder zichtbaar te maken.
        headerFooterManager.SetSlideNumberVisibility(true);
    }
	// Eigenschap IsDateTimeVisible wordt gebruikt om aan te geven dat een slide-datum-tijd-placeholder niet aanwezig is.
    if (!headerFooterManager.IsDateTimeVisible)
    {
		// Methode SetFooterVisibility wordt gebruikt om een slide-datum-tijd-placeholder zichtbaar te maken.
        headerFooterManager.SetDateTimeVisibility(true);
    }
	// Methode SetFooterText wordt gebruikt om tekst in te stellen voor de slide-voettekst-placeholder.
    headerFooterManager.SetFooterText("Footer text");
	// Methode SetDateTimeText wordt gebruikt om tekst in te stellen voor de slide-datum-tijd-placeholder.
    headerFooterManager.SetDateTimeText("Date and time text");
	presentation.Save("Presentation.ppt",SaveFormat.ppt);
}
```
