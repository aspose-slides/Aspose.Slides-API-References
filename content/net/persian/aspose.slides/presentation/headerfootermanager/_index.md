### مثال‌ها

مثال زیر نشان می‌دهد چگونه قابلیت نمایش پاورقی را در اسلاید یک ارائهٔ PowerPoint تنظیم کنید.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IBaseSlideHeaderFooterManager headerFooterManager = presentation.Slides[0].HeaderFooterManager;
	// ویژگی IsFooterVisible برای نشان دادن اینکه جای‌دار فوتر اسلاید حضور ندارد استفاده می‌شود.
    if (!headerFooterManager.IsFooterVisible)
    {
		// متد SetFooterVisibility برای قابل مشاهده کردن جای‌دار فوتر اسلاید استفاده می‌شود.
        headerFooterManager.SetFooterVisibility(true);
    }
	// ویژگی IsSlideNumberVisible برای نشان دادن عدم وجود جای‌دار شماره صفحه اسلاید استفاده می‌شود.
    if (!headerFooterManager.IsSlideNumberVisible)
    {
		// متد SetSlideNumberVisibility برای قابل مشاهده کردن جای‌دار شماره صفحه اسلاید استفاده می‌شود.
        headerFooterManager.SetSlideNumberVisibility(true);
    }
	// ویژگی IsDateTimeVisible برای نشان دادن عدم وجود جای‌دار تاریخ-زمان اسلاید استفاده می‌شود.
    if (!headerFooterManager.IsDateTimeVisible)
    {
		// متد SetFooterVisibility برای قابل مشاهده کردن جای‌دار تاریخ-زمان اسلاید استفاده می‌شود.
        headerFooterManager.SetDateTimeVisibility(true);
    }
	// متد SetFooterText برای تنظیم متن در جای‌دار فوتر اسلاید استفاده می‌شود.
    headerFooterManager.SetFooterText("Footer text");
	// متد SetDateTimeText برای تنظیم متن در جای‌دار تاریخ-زمان اسلاید استفاده می‌شود.
    headerFooterManager.SetDateTimeText("Date and time text");
	presentation.Save("Presentation.ppt",SaveFormat.ppt);
}
```
