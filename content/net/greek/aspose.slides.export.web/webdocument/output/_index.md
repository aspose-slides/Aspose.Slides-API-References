### Παραδείγματα

```csharp
[C#]
using (Presentation pres = new Presentation("pres.pptx"))
{
    var options = new WebDocumentOptions
    {
        TemplateEngine = new RazorTemplateEngine(),
        OutputSaver = new FileOutputSaver(),
        EmbedImages = false
    };
    
    WebDocument document = new WebDocument(options);

    // προσθέστε το "index.html" στα αρχεία εξόδου, χρησιμοποιώντας το πρότυπο "index" για να το δημιουργήσετε και τη μεταβλητή pres ως μοντέλο
    document.Output.Add("index.html", "index", pres);

    // ... ρυθμίστε άλλες επιλογές του εγγράφου και στη συνέχεια αποθηκεύστε το έγγραφο
    document.Save();
}
```
