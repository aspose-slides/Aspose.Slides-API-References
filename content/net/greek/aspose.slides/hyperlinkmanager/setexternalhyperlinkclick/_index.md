## HyperlinkManager.SetExternalHyperlinkClick μέθοδος

### Παραδείγματα

Ο παρακάτω κώδικας δείγματος δείχνει πώς να προσθέσετε ένα πλαίσιο κειμένου με υπερσύνδεσμο.

```csharp
[C#]
// Δημιουργεί ένα αντικείμενο Presentation που αντιπροσωπεύει ένα PPTX
using(Presentation pptxPresentation = new Presentation()) {
  // Αποκτά τη πρώτη διαφάνεια στην παρουσίαση
  ISlide slide = pptxPresentation.Slides[0];
  // Προσθέτει ένα αντικείμενο AutoShape με τύπο Rectangle
  IShape pptxShape = slide.Shapes.AddAutoShape(ShapeType.Rectangle, 150, 150, 150, 50);
  // Κάνει μετατροπή του σχήματος σε AutoShape
  IAutoShape pptxAutoShape = (IAutoShape) pptxShape;
  // Πρόσβαση στην ιδιότητα ITextFrame που συσχετίζεται με το AutoShape
  pptxAutoShape.AddTextFrame("");
  ITextFrame textFrame = pptxAutoShape.TextFrame;
  // Προσθέτει κείμενο στο πλαίσιο
  textFrame.Paragraphs[0].Portions[0].Text = "Aspose.Slides";
  // Ορίζει τον Hyperlink για το κείμενο του τμήματος
  IHyperlinkManager HypMan = textFrame.Paragraphs[0].Portions[0].PortionFormat.HyperlinkManager;
  HypMan.SetExternalHyperlinkClick("http://www.aspose.com");
  // Αποθηκεύει την παρουσίαση PPTX
  pptxPresentation.Save("hLinkPPTX_out.pptx", Aspose.Slides.Export.SaveFormat.Pptx);
}
```
