public IVideoFrame InsertVideoFrame(int index, float x, float y, float width, float height, 
    string fname)

| index | Int32 | Ο δείκτης μηδενικής βάσης όπου θα εισαχθεί το πλαίσιο βίντεο. |
| x | Single | Η συντεταγμένη x του νέου πλαισίου βίντεο, σε μονάδες σημείου. |
