### Παραδείγματα

Ο παρακάτω κώδικας δείγματος δείχνει πώς να ευθυγραμμίσετε παραγράφους κειμένου σε παρουσίαση PowerPoint.

```csharp
[C#]
// Δημιουργεί ένα αντικείμενο Presentation που αντιπροσωπεύει ένα αρχείο PPTX
using (Presentation pres = new Presentation("ParagraphsAlignment.pptx"))
{
    // Πρόσβαση στην πρώτη διαφάνεια
    ISlide slide = pres.Slides[0];
    // Πρόσβαση στο πρώτο και δεύτερο placeholder στη διαφάνεια και μετατροπή του σε AutoShape
    ITextFrame tf1 = ((IAutoShape)slide.Shapes[0]).TextFrame;
    ITextFrame tf2 = ((IAutoShape)slide.Shapes[1]).TextFrame;
    // Αλλαγή του κειμένου και στις δύο placeholders
    tf1.Text = "Center Align by Aspose";
    tf2.Text = "Center Align by Aspose";
    // Λήψη της πρώτης παραγράφου των placeholders
    IParagraph para1 = tf1.Paragraphs[0];
    IParagraph para2 = tf2.Paragraphs[0];
    // Στοίχιση της παραγράφου κειμένου στο κέντρο
    para1.ParagraphFormat.Alignment = TextAlignment.Center;
    para2.ParagraphFormat.Alignment = TextAlignment.Center;
    //Γράψιμο της παρουσίασης ως αρχείο PPTX
    pres.Save("Centeralign_out.pptx", SaveFormat.Pptx);
}
```
