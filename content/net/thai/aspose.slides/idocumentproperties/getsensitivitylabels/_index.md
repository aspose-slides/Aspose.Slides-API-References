## IDocumentProperties.GetSensitivityLabels เมธอด

ดึงอาเรย์ของป้ายความอ่อนไหวจากคุณสมบัติเอกสารที่กำหนดเอง (Microsoft Information Protection SDK Metadata).

```csharp
public ISensitivityLabel[] GetSensitivityLabels()
```

### ตัวอย่าง

โค้ดต่อไปนี้แสดงวิธีการย้ายข้อมูลป้ายความอ่อนไหวจากคุณสมบัติเอกสารที่กำหนดเองไปยังคอลเลกชัน SensitivityLabels สมัยใหม่:

```csharp
[C#]
using (Presentation pres = new Presentation("SomePresentation.pptx"))
{
    // ดึงป้ายความอ่อนไหวจากคุณสมบัติเอกสารที่กำหนดเอง
    ISensitivityLabel[] mipSensitivityLabels = pres.DocumentProperties.GetSensitivityLabels();

    ISensitivityLabelCollection sensitivityLabels = pres.SensitivityLabels;
    foreach (var sensitivityLabel in mipSensitivityLabels)
    {
        // เพิ่มป้ายไปยังคอลเลกชัน
        // ที่นี่คุณสามารถเพิ่มการตรวจสอบความถูกต้องของข้อมูลป้าย (ป้ายมีอยู่ เป็นต้น)
        sensitivityLabels.Add(sensitivityLabel);
    }

    pres.Save("SensitivityLabel.pptx", SaveFormat.Pptx);
}
```
