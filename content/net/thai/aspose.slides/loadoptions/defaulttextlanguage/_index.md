## LoadOptions.DefaultTextLanguage คุณสมบัติ

คืนค่า याกำหนดภาษาดีฟอลต์สำหรับข้อความในงานนำเสนอ. อ่าน/เขียน String.

```csharp
public string DefaultTextLanguage { get; set; }
```

### ตัวอย่าง

ตัวอย่าง:

```csharp
[C#]
// ใช้ตัวเลือกการโหลดเพื่อกำหนดภาษาข้อความเริ่มต้น
LoadOptions loadOptions = new LoadOptions();
loadOptions.DefaultTextLanguage = "en-US";
using (Presentation pres = new Presentation(loadOptions))
 {
    // เพิ่มรูปร่างสี่เหลี่ยมใหม่พร้อมข้อความ
    IAutoShape shp = pres.Slides[0].Shapes.AddAutoShape(ShapeType.Rectangle, 50, 50, 150, 50);
    shp.TextFrame.Text = "New Text";

    // ตรวจสอบภาษาของส่วนแรก
    Console.WriteLine(shp.TextFrame.Paragraphs[0].Portions[0].PortionFormat.LanguageId);
}
```
