## LoadOptions.DefaultRegularFont คุณสมบัติ

คืนค่า หรือ ตั้งค่า ฟอนต์ปกติที่ใช้ในกรณีที่ไม่พบฟอนต์ต้นฉบับ. อ่าน/เขียน String.

```csharp
public string DefaultRegularFont { get; set; }
```

### ตัวอย่าง

ตัวอย่างต่อไปนี้แสดงวิธีตั้งค่าฟอนต์เริ่มต้นสำหรับการเรนเดอร์ PowerPoint Presentation.

```csharp
[C#]
// ใช้ load options เพื่อกำหนดฟอนต์ปกติและฟอนต์เอเชียเริ่มต้น
LoadOptions loadOptions = new LoadOptions(LoadFormat.Auto);
loadOptions.DefaultRegularFont = "Wingdings";
loadOptions.DefaultAsianFont = "Wingdings";
// โหลดพรีเซนเทชัน
using (Presentation pptx = new Presentation("DefaultFonts.pptx", loadOptions))
{
    // สร้างภาพย่อของสไลด์
    pptx.Slides[0].GetThumbnail(1, 1).Save("output_out.png", ImageFormat.Png);
    // สร้าง PDF
    pptx.Save("output_out.pdf", SaveFormat.Pdf);
    // สร้าง XPS
    pptx.Save("output_out.xps", SaveFormat.Xps);
}
```
