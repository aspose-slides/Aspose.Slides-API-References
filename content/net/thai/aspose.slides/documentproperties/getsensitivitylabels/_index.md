## DocumentProperties.GetSensitivityLabels เมธอด

รับอาร์เรย์ของป้ายความอ่อนไหวจากคุณสมบัติเขียนเองของเอกสาร (Microsoft Information Protection SDK Metadata).

```csharp
public ISensitivityLabel[] GetSensitivityLabels()
```

### ตัวอย่าง

โค้ดต่อไปนี้แสดงวิธีย้ายข้อมูลป้ายความอ่อนไหวจากคุณสมบัติเขียนเองของเอกสารไปยังคอลเลกชัน SensitivityLabels สมัยใหม่:

```csharp
[C#]
using (Presentation pres = new Presentation("SomePresentation.pptx"))
{
    // ดึงป้ายความอ่อนไหวจากคุณสมบัติเขียนเองของเอกสาร
    ISensitivityLabel[] mipSensitivityLabels = pres.DocumentProperties.GetSensitivityLabels();

    ISensitivityLabelCollection sensitivityLabels = pres.SensitivityLabels;
    foreach (var sensitivityLabel in mipSensitivityLabels)
    {
        // เพิ่มป้ายลงในคอลเลกชัน
        // ที่นี่คุณสามารถเพิ่มการตรวจสอบความถูกต้องของข้อมูลป้าย (ป้ายพร้อมใช้งาน เป็นต้น)
        sensitivityLabels.Add(sensitivityLabel);
    }

    pres.Save("SensitivityLabel.pptx", SaveFormat.Pptx);
}
```
