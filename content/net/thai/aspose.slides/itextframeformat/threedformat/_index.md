## ITextFrameFormat.ThreeDFormat คุณสมบัติ

คืนค่าอ็อบเจ็กต์ ThreeDFormat ที่แสดงคุณสมบัติของเอฟเฟกต์ 3d สำหรับข้อความ. อ่านอย่างเดียว [`IThreeDFormat`](../../ithreedformat).

```csharp
public IThreeDFormat ThreeDFormat { get; }
```

### ตัวอย่าง

```csharp
[C#]
using (Presentation pres = new Presentation())
{
    IAutoShape autoShape = pres.Slides[0].Shapes.AddAutoShape(ShapeType.Rectangle, 10, 20, 400, 300);

    ITextFrame textFrame = autoShape.TextFrame;

    textFrame.Text = "Aspose.Slide Test Text";

    // ตั้งค่าการแปลงข้อความ
    textFrame.TextFrameFormat.Transform = TextShapeType.ArchUpPour;

    // ตั้งค่าการดันออก
    textFrame.TextFrameFormat.ThreeDFormat.ExtrusionColor.Color = Color.Orange;
    textFrame.TextFrameFormat.ThreeDFormat.ExtrusionHeight = 6;

    // ตั้งค่าขอบ
    textFrame.TextFrameFormat.ThreeDFormat.ContourColor.Color = Color.DarkRed;
    textFrame.TextFrameFormat.ThreeDFormat.ContourWidth = 1.5;

    // ตั้งค่าความลึก
    textFrame.TextFrameFormat.ThreeDFormat.Depth = 3;

    // ตั้งค่าวัสดุ
    textFrame.TextFrameFormat.ThreeDFormat.Material = MaterialPresetType.Plastic;

    // ตั้งค่าแสง
    textFrame.TextFrameFormat.ThreeDFormat.LightRig.Direction = LightingDirection.Top;
    textFrame.TextFrameFormat.ThreeDFormat.LightRig.LightType = LightRigPresetType.Balanced;
    textFrame.TextFrameFormat.ThreeDFormat.LightRig.SetRotation(0, 0, 40);

    // ตั้งค่าประเภทกล้อง
    textFrame.TextFrameFormat.ThreeDFormat.Camera.CameraType = CameraPresetType.PerspectiveContrastingRightFacing;
}
```
