## Presentation.Slides คุณสมบัติ

ส่งคืนรายการของสไลด์ทั้งหมดที่กำหนดไว้ในงานนำเสนอ. อ่านอย่างเดียว [`ISlideCollection`](../th/../islidecollection).

```csharp
public ISlideCollection Slides { get; }
```

### ตัวอย่าง

ตัวอย่างต่อไปนี้แสดงวิธีตั้งค่าสีพื้นหลังของสไลด์ใน PowerPoint Presentation.

```csharp
[C#]
// สร้างอินสแตนซ์ของคลาส Presentation ที่เป็นตัวแทนไฟล์งานนำเสนอ
using (Presentation pres = new Presentation())
{
    // ตั้งค่าสีพื้นหลังของ ISlide แรกเป็นสีน้ำเงิน
    pres.Slides[0].Background.Type = BackgroundType.OwnBackground;
    pres.Slides[0].Background.FillFormat.FillType = FillType.Solid;
    pres.Slides[0].Background.FillFormat.SolidFillColor.Color = Color.Blue;
    pres.Save("ContentBG_out.pptx", SaveFormat.Pptx);
}
```

ตัวอย่างต่อไปนี้แสดงวิธีตั้งค่าภาพพื้นหลังของสไลด์ใน PowerPoint Presentation.

```csharp
[C#]
// สร้างอินสแตนซ์ของคลาส Presentation ที่เป็นตัวแทนไฟล์งานนำเสนอ
using (Presentation pres = new Presentation("SetImageAsBackground.pptx"))
{
    // ตั้งค่าพื้นหลังด้วยรูปภาพ
    pres.Slides[0].Background.Type = BackgroundType.OwnBackground;
    pres.Slides[0].Background.FillFormat.FillType = FillType.Picture;
    pres.Slides[0].Background.FillFormat.PictureFillFormat.PictureFillMode = PictureFillMode.Stretch;
    // ตั้งค่าภาพ
    System.Drawing.Image img = (System.Drawing.Image)new Bitmap(dataDir + "Tulips.jpg");
    // เพิ่มรูปภาพไปยังคอลเลกชันรูปภาพของงานนำเสนอ
    IPPImage imgx = pres.Images.AddImage(img);
    pres.Slides[0].Background.FillFormat.PictureFillFormat.Picture.Image = imgx;
    // เขียนงานนำเสนอไปยังดิสก์
    pres.Save("ContentBG_Img_out.pptx", SaveFormat.Pptx);
}
```

ตัวอย่างต่อไปนี้แสดงวิธีเพิ่มการเปลี่ยนสไลด์ใน Presentation.

```csharp
[C#]
// สร้างอินสแตนซ์ของคลาส Presentation เพื่อโหลดไฟล์งานนำเสนอต้นฉบับ
using (Presentation presentation = new Presentation("AccessSlides.pptx"))
{
    // ใช้การเปลี่ยนสไลด์แบบวงกลมบนสไลด์ที่ 1
    presentation.Slides[0].SlideShowTransition.Type = TransitionType.Circle;
    // ใช้การเปลี่ยนสไลด์แบบคอมบบนสไลด์ที่ 2
    presentation.Slides[1].SlideShowTransition.Type = TransitionType.Comb;
    // เขียนงานนำเสนอไปยังดิสก์
    presentation.Save("SampleTransition_out.pptx", SaveFormat.Pptx);
}
```

ตัวอย่างต่อไปนี้แสดงวิธีเพิ่มการเปลี่ยนสไลด์ขั้นสูง.

```csharp
[C#]
// สร้างอินสแตนซ์ของคลาส Presentation ที่เป็นตัวแทนไฟล์งานนำเสนอ
using (Presentation pres = new Presentation("BetterSlideTransitions.pptx"))
{
    // ใช้การเปลี่ยนสไลด์แบบวงกลมบนสไลด์ที่ 1
    pres.Slides[0].SlideShowTransition.Type = TransitionType.Circle;
    // ตั้งค่าเวลาเปลี่ยนสไลด์เป็น 3 วินาที
    pres.Slides[0].SlideShowTransition.AdvanceOnClick = true;
    pres.Slides[0].SlideShowTransition.AdvanceAfterTime = 3000;
    // ใช้การเปลี่ยนสไลด์แบบคอมบบนสไลด์ที่ 2
    pres.Slides[1].SlideShowTransition.Type = TransitionType.Comb;
    // ตั้งค่าเวลาเปลี่ยนสไลด์เป็น 5 วินาที
    pres.Slides[1].SlideShowTransition.AdvanceOnClick = true;
    pres.Slides[1].SlideShowTransition.AdvanceAfterTime = 5000;
    // ใช้การเปลี่ยนสไลด์แบบซูมบนสไลด์ที่ 3
    pres.Slides[2].SlideShowTransition.Type = TransitionType.Zoom;
    // ตั้งค่าเวลาเปลี่ยนสไลด์เป็น 7 วินาที
    pres.Slides[2].SlideShowTransition.AdvanceOnClick = true;
    pres.Slides[2].SlideShowTransition.AdvanceAfterTime = 7000;
    // เขียนงานนำเสนอไปยังดิสก์
    pres.Save("SampleTransition_out.pptx", SaveFormat.Pptx);
}
```
