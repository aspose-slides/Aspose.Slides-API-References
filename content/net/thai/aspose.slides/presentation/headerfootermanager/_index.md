### ตัวอย่าง

ตัวอย่างต่อไปนี้แสดงวิธีตั้งค่าการมองเห็นส่วนท้ายภายใน Slide ของงานนำเสนอ PowerPoint.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IBaseSlideHeaderFooterManager headerFooterManager = presentation.Slides[0].HeaderFooterManager;
	// คุณสมบัติ IsFooterVisible ใช้เพื่อบ่งชี้ว่าตัวจัดตำแหน่งส่วนท้ายของสไลด์ไม่มีอยู่
    if (!headerFooterManager.IsFooterVisible)
    {
		// เมธอด SetFooterVisibility ใช้เพื่อทำให้ตัวจัดตำแหน่งส่วนท้ายของสไลด์ปรากฏ
        headerFooterManager.SetFooterVisibility(true);
    }
	// คุณสมบัติ IsSlideNumberVisible ใช้เพื่อบ่งชี้ว่าตัวจัดตำแหน่งเลขหน้าของสไลด์ไม่มีอยู่
    if (!headerFooterManager.IsSlideNumberVisible)
    {
		// เมธอด SetSlideNumberVisibility ใช้เพื่อทำให้ตัวจัดตำแหน่งเลขหน้าของสไลด์ปรากฏ
        headerFooterManager.SetSlideNumberVisibility(true);
    }
	// คุณสมบัติ IsDateTimeVisible ใช้เพื่อบ่งชี้ว่าตัวจัดตำแหน่งวันที่และเวลาในสไลด์ไม่มีอยู่
    if (!headerFooterManager.IsDateTimeVisible)
    {
		// เมธอด SetFooterVisibility ใช้เพื่อทำให้ตัวจัดตำแหน่งวันที่และเวลาในสไลด์ปรากฏ
        headerFooterManager.SetDateTimeVisibility(true);
    }
	// เมธอด SetFooterText ใช้เพื่อกำหนดข้อความให้กับตำแหน่งส่วนท้ายของสไลด์
    headerFooterManager.SetFooterText("Footer text");
	// เมธอด SetDateTimeText ใช้เพื่อกำหนดข้อความให้กับตำแหน่งวันที่และเวลาในสไลด์
    headerFooterManager.SetDateTimeText("Date and time text");
	presentation.Save("Presentation.ppt",SaveFormat.ppt);
}
```

ตัวอย่างต่อไปนี้แสดงวิธีตั้งค่าการมองเห็นส่วนท้ายลูกภายใน Slide.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IMasterSlideHeaderFooterManager headerFooterManager = presentation.Masters[0].HeaderFooterManager;
 // Method SetFooterAndChildFootersVisibility is used for making a master slide and all child footer placeholders visible.
    headerFooterManager.SetFooterAndChildFootersVisibility(true);
 // Method SetSlideNumberAndChildSlideNumbersVisibility is used for making a master slide and all child page number placeholders visible.
    headerFooterManager.SetSlideNumberAndChildSlideNumbersVisibility(true);
 // Method SetDateTimeAndChildDateTimesVisibility is used for making a master slide and all child date-time placeholders visible.
    headerFooterManager.SetDateTimeAndChildDateTimesVisibility(true);
 // Method SetFooterAndChildFootersText is used for setting text to master slide and all child footer placeholders.
    headerFooterManager.SetFooterAndChildFootersText("Footer text");
 // Method SetDateTimeAndChildDateTimesText is used for setting text to master slide and all child date-time placeholders.
    headerFooterManager.SetDateTimeAndChildDateTimesText("Date and time text");
}
```
