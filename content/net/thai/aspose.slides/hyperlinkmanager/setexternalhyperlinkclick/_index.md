## HyperlinkManager.SetExternalHyperlinkClick เมธอด

ตั้งค่าลิงก์ภายนอกเมื่อคลิก.

```csharp
public IHyperlink SetExternalHyperlinkClick(string url)
```

| พารามิเตอร์ | ประเภท | คำอธิบาย |
| --- | --- | --- |
| url | String | URL ของไฮเปอร์ลิงก์. |

### ตัวอย่าง

โค้ดตัวอย่างต่อไปนี้แสดงวิธีการเพิ่มกล่องข้อความพร้อมไฮเปอร์ลิงก์.

```csharp
[C#]
// สร้างอินสแตนซ์ของคลาส Presentation ที่แสดงถึงไฟล์ PPTX
using(Presentation pptxPresentation = new Presentation()) {
  // ดึงสไลด์แรกในงานนำเสนอ
  ISlide slide = pptxPresentation.Slides[0];
  // เพิ่มอ็อบเจกต์ AutoShape ที่ประเภทตั้งเป็น Rectangle
  IShape pptxShape = slide.Shapes.AddAutoShape(ShapeType.Rectangle, 150, 150, 150, 50);
  // แคสต์รูปร่างเป็น AutoShape
  IAutoShape pptxAutoShape = (IAutoShape) pptxShape;
  // เข้าถึงคุณสมบัติ ITextFrame ที่เชื่อมโยงกับ AutoShape
  pptxAutoShape.AddTextFrame("");
  ITextFrame textFrame = pptxAutoShape.TextFrame;
  // เพิ่มข้อความบางส่วนลงในเฟรม
  textFrame.Paragraphs[0].Portions[0].Text = "Aspose.Slides";
  // ตั้งค่าลิงก์ไฮเปอร์ลิงก์สำหรับข้อความส่วน
  IHyperlinkManager HypMan = textFrame.Paragraphs[0].Portions[0].PortionFormat.HyperlinkManager;
  HypMan.SetExternalHyperlinkClick("http://www.aspose.com");
  // บันทึกงานนำเสนอ PPTX
  pptxPresentation.Save("hLinkPPTX_out.pptx", Aspose.Slides.Export.SaveFormat.Pptx);
}
```
