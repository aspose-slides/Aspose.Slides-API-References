## WebDocument.Save metódus

Elmenti a dokumentum kimenetét.

```csharp
public void Save()
```

### Példák

```csharp
[C#]        
using (Presentation pres = new Presentation("pres.pptx"))
{
    var options = new WebDocumentOptions
    {
        TemplateEngine = new RazorTemplateEngine(),
        OutputSaver = new FileOutputSaver(),
        EmbedImages = false
    };
    
    WebDocument document = new WebDocument(options);

    // adjunk hozzá egy "index-template.html" sablont az "index" sablonkulccsal, hogy később (a kimenethez) használhassuk
    document.Input.AddTemplate<Aspose.Slides.Presentation>("index", "index-template.html");

    // adjunk hozzá egy "index.html" fájlt a kimeneti fájlokhoz, az "index" sablont használva a generáláshoz és a pres változót modellként
    document.Output.Add("index.html", "index", pres);
    
    // ... állítsuk be a dokumentum egyéb beállításait, majd mentsük a dokumentumot
    document.Global.Put("slideMargin", 10);
    document.Global.Put("imagesPath", "root/site/images");
    // ...
 
    document.Save();
}
```
