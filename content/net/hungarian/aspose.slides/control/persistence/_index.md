```csharp
public PersistenceType Persistence { get; }
```

```csharp
[C#]

    control.Properties["Value"] = value;
}

{
    YourMethodHere(control.ActiveXControlBinary); //Használja a saját módszerét az ActiveX tulajdonságok bináris fájlban tárolt kezeléséhez
```
