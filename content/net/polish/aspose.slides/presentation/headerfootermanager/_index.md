### Przykłady

Poniższy przykład pokazuje, jak ustawić widoczność stopki wewnątrz Slide w prezentacji PowerPoint.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IBaseSlideHeaderFooterManager headerFooterManager = presentation.Slides[0].HeaderFooterManager;
	// Właściwość IsFooterVisible jest używana do wskazywania, że placeholder stopki slajdu nie jest obecny.
    if (!headerFooterManager.IsFooterVisible)
    {
		// Metoda SetFooterVisibility jest używana do ustawienia widoczności placeholdera stopki slajdu.
        headerFooterManager.SetFooterVisibility(true);
    }
	// Właściwość IsSlideNumberVisible jest używana do wskazywania, że placeholder numeru slajdu nie jest obecny.
    if (!headerFooterManager.IsSlideNumberVisible)
    {
		// Metoda SetSlideNumberVisibility jest używana do ustawienia widoczności placeholdera numeru slajdu.
        headerFooterManager.SetSlideNumberVisibility(true);
    }
	// Właściwość IsDateTimeVisible jest używana do wskazywania, że placeholder daty i godziny slajdu nie jest obecny.
    if (!headerFooterManager.IsDateTimeVisible)
    {
		// Metoda SetFooterVisibility jest używana do ustawienia widoczności placeholdera daty i godziny slajdu.
        headerFooterManager.SetDateTimeVisibility(true);
    }
	// Metoda SetFooterText jest używana do ustawiania tekstu w placeholderze stopki slajdu.
    headerFooterManager.SetFooterText("Footer text");
	// Metoda SetDateTimeText jest używana do ustawiania tekstu w placeholderze daty i godziny slajdu.
    headerFooterManager.SetDateTimeText("Date and time text");
	presentation.Save("Presentation.ppt",SaveFormat.ppt);
}
```
