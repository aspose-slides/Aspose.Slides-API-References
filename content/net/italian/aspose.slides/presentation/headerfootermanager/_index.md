### Esempi

Il seguente esempio mostra come impostare la visibilità del footer all'interno della Slide di una presentazione PowerPoint.

```csharp
[C#]
using (Presentation presentation = new Presentation("presentation.ppt"))
{
    IBaseSlideHeaderFooterManager headerFooterManager = presentation.Slides[0].HeaderFooterManager;
	// La proprietà IsFooterVisible è usata per indicare che il segnaposto del footer della diapositiva non è presente.
    if (!headerFooterManager.IsFooterVisible)
    {
		// Il metodo SetFooterVisibility è usato per rendere visibile il segnaposto del footer della diapositiva.
        headerFooterManager.SetFooterVisibility(true);
    }
	// La proprietà IsSlideNumberVisible è usata per indicare che il segnaposto del numero di pagina della diapositiva non è presente.
    if (!headerFooterManager.IsSlideNumberVisible)
    {
		// Il metodo SetSlideNumberVisibility è usato per rendere visibile il segnaposto del numero di pagina della diapositiva.
        headerFooterManager.SetSlideNumberVisibility(true);
    }
	// La proprietà IsDateTimeVisible è usata per indicare che il segnaposto della data e ora della diapositiva non è presente.
    if (!headerFooterManager.IsDateTimeVisible)
    {
		// Il metodo SetFooterVisibility è usato per rendere visibile il segnaposto della data e ora della diapositiva.
        headerFooterManager.SetDateTimeVisibility(true);
    }
	// Il metodo SetFooterText è usato per impostare il testo nel segnaposto del footer della diapositiva.
    headerFooterManager.SetFooterText("Footer text");
	// Il metodo SetDateTimeText è usato per impostare il testo nel segnaposto della data e ora della diapositiva.
    headerFooterManager.SetDateTimeText("Date and time text");
	presentation.Save("Presentation.ppt",SaveFormat.ppt);
}
```

	// Il metodo SetFooterAndChildFootersVisibility è usato per rendere visibili il master slide e tutti i segnaposto footer dei figli.

	// Il metodo SetSlideNumberAndChildSlideNumbersVisibility è usato per rendere visibili il master slide e tutti i segnaposto del numero di pagina dei figli.
    headerFooterManager.SetSlideNumberAndChildSlideNumbersVisibility(true);
	// Il metodo SetDateTimeAndChildDateTimesVisibility è usato per rendere visibili il master slide e tutti i segnaposto data/ora dei figli.
    headerFooterManager.SetDateTimeAndChildDateTimesVisibility(true);
	// Il metodo SetFooterAndChildFootersText è usato per impostare il testo nel master slide e in tutti i segnaposto footer dei figli.
    headerFooterManager.SetFooterAndChildFootersText("Footer text");
	// Il metodo SetDateTimeAndChildDateTimesText è usato per impostare il testo nel master slide e in tutti i segnaposto data/ora dei figli.
    headerFooterManager.SetDateTimeAndChildDateTimesText("Date and time text");
}
