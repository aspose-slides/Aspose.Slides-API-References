## PresentationPlayer class

 Represents the player of animations associated with the  Presentation. 
 

 
```php
  $pres = new Presentation("pres.pptx");
  try {
    $animationsGenerator = new PresentationAnimationsGenerator($pres);
    try {
      // Play animation with 33 FPS
      $player33 = new PresentationPlayer($animationsGenerator, 33);
      try {
        $player33->setFrameTick(( sender, args) -> {
          try {
            ImageIO->write($args->getFrame(), "PNG", new java.io.File("33fps/frame_" + $sender->getFrameIndex() + ".png"));
          } catch (JavaException $e) {
            throw new <e>RuntimeException();
          }
        });
        $animationsGenerator->run($pres->getSlides());
      } finally {
        if ($player33 != null) {
          $player33->dispose();
        }
      }
      // Play animation with 45 FPS
      $player45 = new PresentationPlayer($animationsGenerator, 45);
      try {
        $player45->setFrameTick(( sender, args) -> {
          try {
            ImageIO->write($args->getFrame(), "PNG", new java.io.File("45fps/frame_" + $sender->getFrameIndex() + ".png"));
          } catch (JavaException $e) {
            throw new <e>RuntimeException();
          }
        });
        $animationsGenerator->run($pres->getSlides());
      } finally {
        if ($player45 != null) {
          $player45->dispose();
        }
      }
    } finally {
      if ($animationsGenerator != null) {
        $animationsGenerator->dispose();
      }
    }
  } finally {
    if ($pres != null) {
      $pres->dispose();
    }
  }
```
